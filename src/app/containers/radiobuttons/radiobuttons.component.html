Radiobutton with ngModel 
    import {Component} from '@angular/core'
    
    @Component({
      selector: 'app-radio-button',
      templateUrl: './app-radio-button.html', 
      styleUrls: ['./app-radio-button.scss'],  
      }) 
    export class RadioButtonComponent   {
    favoriteSeason: string;
    seasons: string[] = ['Winter', 'Spring', 'Summer', 'Autumn'];
      }
        
    <neo-radio-group 
      aria-labelledby="example-radio-group-label"
      class="example-radio-group"
      [(ngModel)]="favoriteSeason"> 
          < neo-radio-button
            class="example-radio-button radio"
            *ngFor="let season of seasons"
            [value]="season">
             { {season} }
          </neo-radio-button >    
    </neo-radio-group >
     <span>Your favorite season is: { {favoriteSeason } }</span>
    
      

        
          .example-radio-group- {    
          display: flex;
          flex-direction: column;
          margin: 15px 0;  
          }
