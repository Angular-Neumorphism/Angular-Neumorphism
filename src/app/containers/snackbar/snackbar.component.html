NeoSnackBar is a service for displaying snack-bar notifications.A snack-bar can contain either a string message or a given component.
Basic snack-bar 
    import {Component} from '@angular/core'
    import { NeoSnackBar } from '@neomorphism/ng-neomorphism/neo-snack-bar';
    
    @Component({
      selector: 'app-snackbar',
      templateUrl: './app-snackbar.html', 
      styleUrls: ['./app-snackbar.scss'],  
      }) 
    export class DialogComponent   {
    constructor(private _snackBar: NeoSnackBar) { }
    openSnackBar(message: string, action: string) {
      this._snackBar.open(message, action, {
        duration: 2000,
        });
        }
        
    <neo-form-field > 
      <neo-lalel > Message</neo-lalel > 
      <input neoInput value="Disco party!" #message / > 
    </neo-form-field > 
    <neo-form-field > 
      <neo-lalel > Action</neo-lalel > 
      <input neoInput value="Dance" #action / > 
    </neo-form-field > 
    
    < button neo-button color='primary' class="button" (click)="openSnackBar(message.value, action.value)">
      Show snack-bar
      < /button>
