Dialog with config object, data receiving and subscription 
    import {Component} from '@angular/core'
    
    @Component({
    selector: 'app-dialog',
    templateUrl: './app-dialog.html', 
    styleUrls: ['./app-dialog.scss'],  
    }) 
  export class DialogComponent   {
  public name: string;
  constructor(private dialog: NeoDialog) { }
  openDialog() {
  const dialogRef = this.dialog.open(DialogPageComponent, {
    width: '250px',
    data: { configurable: true, name: undefined },
    });
    dialogRef.afterClosed().subscribe((data) => {
      if (data.name) {
        this.name = data.name;
        }
      });
    }
    }

  @Component({
  selector: 'app-dialog-page',
  templateUrl: './app-dialog-page.html', 
  styleUrls: ['./app-dialog-page.scss'],  
  }) 
export class DialogPageComponent   {

  }

Directives

NeoDialogClose

Button that will close the current dialog. Selector: [neo-dialog-close] [neoDialogClose] Exported as: matDialogClose

NeoDialogTitle

Title of a dialog element. Stays fixed to the top of the dialog when scrolling. Selector: [neo-dialog-title] [neoDialogTitle] Exported as: matDialogTitle

NeoDialogContent

Scrollable content container of a dialog. Selector: [neo-dialog-content] neo-dialog-content [neoDialogContent]

NeoDialogActions

Container for the bottom action buttons in a dialog. Stays fixed to the bottom when scrolling. Selector: [neo-dialog-actions] neo-dialog-actions [neoDialogActions]
